    <h1>Instagram Reel View Counter</h1>

    <form id="reelForm">

        <label for="reelUrl">https://www.instagram.com/p/DAaihQ4REEE/?hl=es:</label>

        <input type="text" id="reelUrl" name="reelUrl" required>

        <button type="submit">Check Views</button>

    </form>


    <h2>View Count: <span id="viewCount">0</span></h2>


    <script>

        const form = document.getElementById('reelForm');

        const viewCountElement = document.getElementById('viewCount');


        form.addEventListener('submit', function(e) {

            e.preventDefault();

            const reelUrl = document.getElementById('reelUrl').value;

            if (reelUrl) {

                fetchViews(reelUrl);

            }

        });


        function fetchViews(reelUrl) {

            setInterval(async function() {

                try {

                    const response = await fetch(`https://tu-api-server.com/get_views?url=${encodeURIComponent(reelUrl)}`);

                    const data = await response.json();

                    if (data.views !== undefined) {

                        viewCountElement.innerText = data.views;

                    } else {

                        viewCountElement.innerText = 'Error fetching views';

                    }

                } catch (error) {

                    console.error('Error fetching views:', error);

                }

            }, 10000); // Actualiza cada 10 segundos

        }

    </script>
